Bar var ctx3=document.getElementById('chartBar').getContext('2d'); chartInstances.push(new Chart(ctx3,{ type:'bar', data:{ labels:['Inflows','Outflows','Profit'], datasets:[{data:[cCont+cDiv,cWith,final-P0-(cCont+cDiv)+cWith],backgroundColor:['#2e7d32','#d32f2f','#1976d2']}] }, options:{responsive:true,maintainAspectRatio:false,plugins:{legend:{display:false}}} })); // 4. Pie var ctx4=document.getElementById('chartPie').getContext('2d'); chartInstances.push(new Chart(ctx4,{ type:'pie', data:{ labels:['Initial','Profit','Net Contrib','Dividends'], datasets:[{data:[P0, final-P0-(cCont+cDiv)+cWith, cCont-cWith, cDiv],backgroundColor:['#0d47a1','#1976d2','#2e7d32','#f9a825']}] }, options:{responsive:true,maintainAspectRatio:false} })); // 5. Waterfall var ctx5=document.getElementById('chartWaterfall').getContext('2d'); var wfData=[P0, cCont, cDiv, -(cWith), final-(P0+cCont+cDiv-cWith)]; var wfLabels=['Start','Contrib','Div','With','Profit']; chartInstances.push(new Chart(ctx5,{ type:'bar', data:{labels:wfLabels,datasets:[{data:wfData,backgroundColor:['#0d47a1','#2e7d32','#f9a825','#d32f2f','#1976d2']}]} })); } /* ===== EXPORT FUNCTIONS ===== */ function printResults(){window.print();} function savePDF(){ var element=document.body; html2canvas(element).then(function(canvas){ var imgData=canvas.toDataURL('image/png'); var pdf=new jspdf.jsPDF('p','mm','a4'); var imgProps=pdf.getImageProperties(imgData); var pdfWidth=pdf.internal.pageSize.getWidth(); var pdfHeight=(imgProps.height*pdfWidth)/imgProps.width; pdf.addImage(imgData,'PNG',0,0,pdfWidth,pdfHeight); pdf.save('Teamwork4Freedom_Results.pdf'); }); } function saveJPG(){ var element=document.body; html2canvas(element).then(function(canvas){ var link=document.createElement('a'); link.download='Teamwork4Freedom_Results.jpg'; link.href=canvas.toDataURL('image/jpeg'); link.click(); }); }
